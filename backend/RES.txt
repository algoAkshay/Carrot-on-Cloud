create table contest_results
(
    contest_id  int                                 not null,
    handle      varchar(100)                        not null,
    performance VARCHAR(10)                         not null,
    delta       VARCHAR(10)                         not null,
    rating      int                                 not null,
    created_at  timestamp default current_timestamp not null,
    updated_at  timestamp default current_timestamp not null on update current_timestamp,
    constraint contest_results_pk
        primary key (contest_id, handle)
);

create index contest_results_handle_contest_id_index
    on contest_results (handle, contest_id);


create table ratingtable
(
    handle varchar(100) not null
        primary key,
    rating int          not null
);

mysql -h <db-host> \
     -u <db-user> \
     -p
